Toast Manager Code

Add this JavaScript to enable the toast notification system used in the Live Demo above:

// Toast Manager for Alpine.js
function toastManager() {
  return {
    toasts: [],
    init() {
      window.addEventListener('show-toast', (e) => {
        this.addToast(e.detail);
      });
    },
    addToast({ type = 'info', title = '', message = '', duration = 4000 }) {
      const id = Date.now();
      this.toasts.push({ id, type, title, message, visible: true });

      setTimeout(() => {
        this.removeToast(id);
      }, duration);
    },
    removeToast(id) {
      const toast = this.toasts.find(t => t.id === id);
      if (toast) {
        toast.visible = false;
        setTimeout(() => {
          this.toasts = this.toasts.filter(t => t.id !== id);
        }, 300);
      }
    }
  };
}

// Usage: Dispatch event from anywhere
