Input States

Disabled, readonly, and focus states

Must be at least 3 characters long.

<!-- Normal -->
<input class="paper-input" type="text" placeholder="Editable input">

<!-- Disabled -->
<input class="paper-input" type="text" placeholder="Cannot edit" disabled>

<!-- Readonly -->
<input class="paper-input bg-gray-50" type="text" value="Readonly value" readonly>

<!-- With helper text -->
<input class="paper-input" type="text" placeholder="Enter username">
<p class="mt-1 text-xs text-[var(--paper-muted)]">Must be at least 3 characters.</p>

  <label for="float-name"
    class="absolute left-3 top-1/2 -translate-y-1/2 text-sm text-[var(--paper-muted)]
           transition-all peer-placeholder-shown:top-1/2 peer-placeholder-shown:text-sm
           peer-focus:top-2 peer-focus:text-xs peer-focus:text-[var(--paper-primary)]

</div>

Textareas

Multi-line text inputs

/200 

<!-- Default -->
<textarea class="paper-textarea" rows="4" placeholder="Enter message..."></textarea>

<!-- Auto-resize -->
<textarea
  class="paper-textarea resize-none overflow-hidden"
  rows="2"
  @input="$el.style.height = 'auto'; $el.style.height = $el.scrollHeight + 'px'"
></textarea>

<!-- With character counter -->
<textarea
  class="paper-textarea"
  maxlength="200"
  @input="charCount = $el.value.length"
></textarea>
<p><span x-text="charCount"></span>/200</p>
