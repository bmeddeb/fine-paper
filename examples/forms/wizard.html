<div x-data="{
  step: 1,
  totalSteps: 3,
  form: { username: '', email: '' },
  canProceed() {
    if (this.step === 1) return this.form.username && this.form.email;
    return true;
  },
  next() { if (this.canProceed()) this.step++; },
  prev() { if (this.step > 1) this.step--; }
}">
  <!-- Step Indicators -->
  <template x-for="i in totalSteps">
    <span :class="step >= i ? 'bg-[var(--wizard-accent)]' : 'bg-[var(--wizard-neutral)]/10'" x-text="i"></span>
  </template>

  <!-- Step Content -->
  <div x-show="step === 1" x-transition>Step 1 content</div>
  <div x-show="step === 2" x-transition>Step 2 content</div>

  <!-- Navigation -->
  <button @click="prev()" x-show="step > 1">Previous</button>
  <button @click="next()" :disabled="!canProceed()">Next</button>
</div>

<div x-data="{
  step: 1,
  totalSteps: 4,
  get progress() { return ((this.step - 1) / (this.totalSteps - 1)) * 100; }
}">
  <!-- Progress Bar -->
  <div class="h-2 bg-[var(--paper-border)] rounded-full">
    <div class="h-full bg-[var(--wizard-accent)] transition-all"
      :style="'width: ' + progress + '%'"></div>
  </div>
</div>

<div x-data="{
  step: 1,
  steps: [
    { icon: 'fa-shopping-cart', title: 'Cart' },
    { icon: 'fa-truck', title: 'Shipping' },
    { icon: 'fa-credit-card', title: 'Payment' },
    { icon: 'fa-check', title: 'Complete' }
  ]
}">
  <template x-for="(s, i) in steps">
    <div class="flex h-14 w-14 items-center justify-center rounded-full"
      :class="step > i + 1 ? 'bg-[var(--wizard-accent)] text-white'
            : step === i + 1 ? 'bg-[var(--wizard-accent)]/15 text-[var(--wizard-accent)] ring-2 ring-[var(--wizard-accent)]'
            : 'bg-[var(--wizard-neutral)]/10 text-[var(--wizard-neutral)]'">
      <i class="fa" :class="step > i + 1 ? 'fa-check' : s.icon"></i>
    </div>
  </template>
</div>

Vertical Wizard

Sidebar step navigation

<div x-data="{ step: 1, steps: ['Info', 'Company', 'Social', 'Done'] }">
  <div class="flex gap-6">
    <!-- Vertical Steps -->
    <div class="w-48">
      <template x-for="(s, i) in steps">
        <button
          class="w-full p-3 rounded-lg"
          :class="step === i + 1 ? 'bg-[var(--wizard-accent)] text-white' : 'bg-[var(--wizard-neutral)]/10 text-[var(--wizard-neutral)]'"
          @click="step = i + 1"
        >
          <span x-text="i + 1"></span>
          <span x-text="s"></span>
        </button>
      </template>
    </div>
    <!-- Content -->
    <div class="flex-1">
      <div x-show="step === 1">Step 1 content</div>
      <div x-show="step === 2">Step 2 content</div>
    </div>
  </div>
</div>
