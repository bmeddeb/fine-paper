x-data="{
  search: '',
  sortCol: 'name',
  sortAsc: true,
  page: 1,
  perPage: 5,
  allData: [...],
  get filtered() {
    let data = this.allData;
    if (this.search) {
      const s = this.search.toLowerCase();
      data = data.filter(r =>
        r.name.toLowerCase().includes(s) ||
        r.position.toLowerCase().includes(s)
      );
    }
    return data;
  },
  get sorted() {
    return [...this.filtered].sort((a, b) => {
      let va = a[this.sortCol], vb = b[this.sortCol];
      if (typeof va === 'string') { va = va.toLowerCase(); vb = vb.toLowerCase(); }
      if (va < vb) return this.sortAsc ? -1 : 1;
      if (va > vb) return this.sortAsc ? 1 : -1;
      return 0;
    });
  },
  get paginated() {
    const start = (this.page - 1) * this.perPage;
    return this.sorted.slice(start, start + this.perPage);
  },
  sort(col) {
    if (this.sortCol === col) this.sortAsc = !this.sortAsc;
    else { this.sortCol = col; this.sortAsc = true; }
  }
}"

x-data="{
  categoryFilter: '',
  statusFilter: '',
  products: [...],
  get categories() { return [...new Set(this.products.map(p => p.category))]; },
  get filtered() {
    return this.products.filter(p => {
      if (this.categoryFilter && p.category !== this.categoryFilter) return false;
      if (this.statusFilter && p.status !== this.statusFilter) return false;
      return true;
    });
  }
}"

<select x-model="categoryFilter">
  <option value="">All Categories</option>
  <template x-for="cat in categories">
    <option :value="cat" x-text="cat"></option>
  </template>
</select>
