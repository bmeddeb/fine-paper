Checkbox Selection

Select multiple items with bulk actions

items selected
Product SKU Price Stock 
<div x-data="{
  items: [...],
  selected: [],
  get allSelected() { return this.selected.length === this.items.length; },
  toggleAll() { this.selected = this.allSelected ? [] : this.items.map(i => i.id); },
  isSelected(id) { return this.selected.includes(id); },
  toggle(id) {
    if (this.isSelected(id)) {
      this.selected = this.selected.filter(i => i !== id);
    } else {
      this.selected.push(id);
    }
  }
}">
  <!-- Select all checkbox in header -->
  <input type="checkbox" :checked="allSelected" @change="toggleAll()">

  <!-- Entry checkbox -->
  <input type="checkbox" :checked="isSelected(item.id)" @change="toggle(item.id)">
</div>
